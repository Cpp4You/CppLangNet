#include <iostream>
#include <string>
#include <vector>

int main()
{
	using std::vector, std::string, std::cout;

	vector<string> playerNames = {
		"HappyBanana",
		"AngryCrab",
		"SadWolf",
	};

	// Adding a new player:
	playerNames.push_back("WickedWitch");

	// Print each player's name
	// highlight-start
	for (auto name : playerNames)
	{
		cout << name << '\n';
	}
	// highlight-end
}
