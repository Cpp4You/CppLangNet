```cpp showLineNumbers
#include <iostream>
#include <string>

struct Vehicle
{
    std::string brand;
    std::string model;
    int productionYear;
};

struct Car : Vehicle
{
    bool isCabrio;
};

struct Airplane : Vehicle
{
    int enginesNumber;
    std::string airlinesName;
};

int main()
{
    Car ford;
    ford.brand = "Ford";
    ford.model = "Fiesta";
    ford.productionYear = 2010;
    ford.isCabrio = false;

    Airplane boeing;
    boeing.brand = "Boeing";
    boeing.model = "737";
    boeing.productionYear = 2010;
    boeing.enginesNumber = 2;
    boeing.airlinesName = "Air Canada";
    
    // highlight-next-line
    std::cout << ford.brand << '\n'; // "Ford"
    // highlight-next-line
    std::cout << boeing.brand << '\n'; // "Boeing"
}
```
