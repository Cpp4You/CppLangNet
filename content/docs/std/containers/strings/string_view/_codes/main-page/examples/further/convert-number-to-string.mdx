```cpp title='Convert numbers to std::string'
#include <iostream>
#include <string>

int main()
{
	int n1		= 123;
	float n2	= 456.20f;
	double n3	= 3.14159;

	std::string s1 = std::to_string(n1); // int -> string
	std::string s2 = std::to_string(n2); // float -> string
	std::string s3 = std::to_string(n3); // double -> string

	std::cout 	<< "s1: " << s1 << '\n'
				<< "s2: " << s2 << '\n'
				<< "s3: " << s3;
}
```
