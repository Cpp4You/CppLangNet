```cpp
#include <array>
#include <string>
#include <istream>

std::string readContents(std::istream& input)
{
	constexpr int BUFFER_SIZE = 16 * 1024;

	std::array<char, BUFFER_SIZE> buffer;
	std::string contents;

	while (input.read(buffer.data(), buffer.size()))
		contents.append(buffer.data(), input.gcount());
	contents.append(buffer.data(), input.gcount());

	return contents;
}
```
